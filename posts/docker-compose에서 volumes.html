tl;dr

👯‍♂️ docker compose에서 volumes

FROM python:3.12.6-slim-workbook

ENV PYTHONDONTWRITEBYTECODE 1
ENV PYTHONUNBUFFERED 1

WORKDIR /usr/src/app

COPY ./requirments .
RUN pip install --upgrade pip & pip install -r requirements.txt 
COPY . .

├── .gitignore
├── docker-compose.yml
└── project
    ├── .dockerignore
    ├── Dockerfile
    ├── app
    │   ├── __init__.py
    │   ├── config.py
    │   └── main.py
    └── requirements.txt

# docker-compose.yml
version: '3.8'

services:
    web:
      build: ./projects
      command: uvicorn --reload app.main:app --workers 1 --host 0.0.0.0 --port 8000
      volumes:
        - ./projects:/usr/src/app
      port:
        - 8080:8000
      enviroments:
        - ENVIRONMENT=dev
        - TESTING=1

The volume is used to mount the code into the container. This is a must for a development environment in order to update the container whenever a change to the source code is made. Without this, you would have to re-build the image each time you make a change to the code
