import random

from functools import lru_cache, cache

@lru_cache(maxsize=128, typed=False)
def cached_fetch_user(user_id:str) -> dict:
    print(f"cached {user_id}")
    return {"user_id": user_id,"email" : f"{user_id}@gmail.com"}

def fetch_user(user_id:str) -> dict:
    print(f"not cached {user_id}")
    return {"user_id": user_id,"email" : f"{user_id}@gmail.com"}


if __name__ == "__main__":
    for _ in range(10):
